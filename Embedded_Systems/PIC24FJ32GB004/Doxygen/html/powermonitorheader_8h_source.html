213 void reset_i2c_bus(void)
214 {
215  int x = 0;
216 
217  //initiate stop bit
218  I2C1CONbits.PEN = 1;
219 
220  //wait for hardware clear of stop bit
221  while (I2C1CONbits.PEN)
222  {
223  __delay_us(1);
224  x ++;
225  if (x > 20) break;
226  }
227  I2C1CONbits.RCEN = 0;
228  IFS1bits.MI2C1IF = 0; // Clear Interrupt
229  I2C1STATbits.IWCOL = 0;
230  I2C1STATbits.BCL = 0;
231  __delay_us(10);

251 {
252  int x = 0;
253  I2C1CONbits.ACKDT = 0; //Reset any previous Ack
254  __delay_us(10);

289 //basic I2C byte send
290 char send_i2c_byte(int data)
291 {
292  int i;
293 
294  while (I2C1STATbits.TBF) { } // if a transmit is in progress, wait
295  IFS1bits.MI2C1IF = 0; // Clear Interrupt
296  I2C1TRN = data; // load the outgoing data byte
297 
298  // wait for transmission
299  for (i=0; i<500; i++)
300  {
301  if (!I2C1STATbits.TRSTAT) break;
302  __delay_us(1);
303 
304  }
305  if (i == 500) {
306  return(1);
307  }
308 
309  // Check for ACK from slave, abort if not found
310  if (I2C1STATbits.ACKSTAT == 1)
311  {
312  // if no acknowledge was found turn on LED
313  //PIN8_LAT = OFF;
314  return(1);
315  }
316  // if the slave acknowledge the LED will turn on
317  //else PIN8_LAT = ON;
318 
319  __delay_us(2);
320  return(0);
321 }
